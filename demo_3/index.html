Tuning a vocabulary layer

      const strings = loadMyStrings();
    

      const vocabInitializer = tf.initializers.knownVocab(
        {strings: ['hello', 'world', 'こんにちは', '世界']});
    

      vocabLayer = tf.layers.vocab({
          name: 'myVocabLayer',
          knownVocabSize: 4,
          hashVocabSize: 1,
          vocabInitializer: vocabInitializer,
          optimizer: new tf.preprocessing.VocabLayerOptimizer(),
          inputShape: [2]  // two words per example
        });
    

      const vocabModel = tf.sequential({
        layers: [vocabLayer]
      });
    

      vocabModel.predict(
        tf.preprocessing.stringTensor(
          [['hello', 'world']])).print();
    

      vocabModel.predict(
        tf.preprocessing.stringTensor(
          [['こんにちは', '世界']])).print();
    

      vocabModel.predict(
        tf.preprocessing.stringTensor(
          [['outOf', 'vocabulary']])).print();
    

      vocabModel.predict(
        tf.preprocessing.stringTensor(
          [['outOf', 'vocabulary']])).print();
    

      vocabModel.compile({optimizer: 'SGD', loss: 'meanSquaredError'});
    

      const trainInputs = tf.preprocessing.stringTensor2d(
        [['a', 'a'], ['b', 'b'], ['c', 'c'], ['d', 'd']]);
    

      // Fit the model to a tensor of strings.
      await vocabModel.fit(trainInputs, null, {batchSize: 1, epochs: 1});
    

      const testInputs = tf.preprocessing.stringTensor2d(
        [['a', 'b'], ['c', 'd'], ['hello', 'world']]);
    

      const testOutputs = vocabModel.predict(testInputs);

Image preprocessing for image classification with zero-mean, unit-std

      const images = loadMyImages();
    

      images.shape;
      images.dtype;
    

      images.min().print();
      images.max().print();
    

      zeroMeanLayer = tf.layers.zeroMean(
        {
           optimizer: new tf.preprocessing.ZeroMeanOptimizer(),
           inputShape: [224, 224, 3]
        });
        

      unitVarLayer = tf.layers.unitVariance(
        {
          optimizer: new tf.preprocessing.UnitVarianceOptimizer()
        });
    

      preprocModel = tf.sequential({
        layers: [ zeroMeanLayer, unitVarLayer ]});
    

      preprocModel.compile({optimizer: 'SGD', loss: 'meanSquaredError'});
    

      await preprocModel.fit(images);
    

      preprocImages = preprocModel.predict(images);
    

      preprocImages.min().print();
      preprocImages.max().print();
    

      sr = await preprocModel.save('downloads://preproc-model');
      sr = await preprocModel.save('indexeddb://preproc-model');
